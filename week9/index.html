        const cookies = document.cookie.split("; ");
        for (crumb of cookies){
          const [key,value] = crumb.split("=");
          console.log(`The value of ${key} is ${value}`);
        }
        << The value of name is Batman
        The value of hero is true
        The value of city is Gotham
